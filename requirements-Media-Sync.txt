# Requirements for Media-Sync.py
#
# Python >= 3.9 required (asyncio features, winrt bindings).
#
# ── Cross-platform ────────────────────────────────────────────────────────────
pystray>=0.19.5
Pillow>=10.0.0

# ── Windows only ──────────────────────────────────────────────────────────────
# The old monolithic 'winrt' package (1.x) is deprecated and lacks pre-built
# binaries for Python 3.9+. The project was split into per-namespace packages
# under the 'pywinrt' umbrella. The packages below cover the full dependency
# chain — pywinrt does NOT pull inter-package deps automatically, so they must
# all be listed explicitly or pip will leave them out.
#
#   winrt-runtime                  – core WinRT async / COM glue
#   winrt-Windows.Foundation       – IAsyncOperation and base WinRT types
#   winrt-Windows.Foundation.Collections – IVector / IMap used by media APIs
#   winrt-Windows.Media            – base media namespace
#   winrt-Windows.Media.Control    – GlobalSystemMediaTransportControlsSession*
#
# The import paths in Media-Sync.py are unchanged — no code edits required.
#
winrt-runtime>=2.0.0; sys_platform == "win32"
winrt-Windows.Foundation>=2.0.0; sys_platform == "win32"
winrt-Windows.Foundation.Collections>=2.0.0; sys_platform == "win32"
winrt-Windows.Media>=2.0.0; sys_platform == "win32"
winrt-Windows.Media.Control>=2.0.0; sys_platform == "win32"

# ── Linux only ────────────────────────────────────────────────────────────────
# PyGObject ('gi' / GTK 3.0) for the system-tray / GTK UI path.
# Prefer your distro's package manager to get the full native stack:
#   Debian / Ubuntu:  sudo apt install python3-gi python3-gi-cairo gir1.2-gtk-3.0
#   Fedora / RHEL:    sudo dnf install python3-gobject gtk3
#   Arch:             sudo pacman -S python-gobject gtk3
#
# pip fallback (requires native GTK libraries to already be installed):
PyGObject>=3.42.0; sys_platform != "win32"

# evdev is optional — Media-Sync detects it at runtime via importlib.
# Install it to enable Linux input-device (keyboard/media-key) monitoring:
evdev>=1.6.0; sys_platform == "linux"
